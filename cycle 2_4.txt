DELIMITER #
CREATE PROCEDURE oddeven(num INT)
BEGIN
DECLARE i INT;
CREATE TABLE even(EVEN_NUMBERS INT);
CREATE TABLE odd(ODD_NUMBERS INT);
SET i=1;
WHILE i<=num DO
IF i%2=0 THEN
INSERT INTO even VALUES(i);
ELSE
INSERT INTO odd VALUES(i);
END IF;
SET i=i+1;
END WHILE;
SELECT * FROM odd;
SELECT * FROM even;
DROP TABLE odd;
DROP TABLE even;
END #

DELIMITER ;
CALL oddeven(10);
